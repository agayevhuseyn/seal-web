// include I/O module
include io

// prints hello world with newline
print('Hello, Seal')

// define factorial function
define factor(n)
    if n == 1 or n == 0
        return 1
    return n * factor(n - 1)

// call the function
result = factor(5)

// and store the result in a file
file = io.File('result.txt', 'w')

file..write(str(result))
// is equivalent to file.write(file, str(result))

// close the file
file..close()
